This is an example of the death_date field in the documents of the notices collection:

death_date
2026-01-02T00:00:00.000+00:00

death_date is of type Date in the NoticesSchema.

In the function below, I need to retrieve only notices where the year of the death_date in Mongodb
equals the year parameter:

exports.getNotices = async (req, res) => {
  console.log("Notices.service.getNotices called...");
  console.log('req.query.year', req.query.year)

  const year = req.query.year;

  try {
    const notices = await Notices.find()
      .sort({ createdAt: "desc" })
      .populate("contacts") // populate contacts
      .populate("events")
      .exec();

    console.log("notices retrieved:", notices);
    return res.status(200).json(notices);
  } catch (error) {
    console.error("Error in getNotices:", error);
    return res.status(500).json({ message: "Internal server error" });
  }
};
